💻 USING RESPONSIVE UTILITIES

import 'package:incidents_managment/core/helpers/responsive.dart';

// Check device type
if (ResponsiveHelper.isMobile(context)) {
  // Single column layout
}

// Responsive font size

  mobileSize: 14,
  desktopSize: 18,
